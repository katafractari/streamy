  152.  152     /* Connect()
  153.  153      * 
  154.  154      * Connects to the MPD server. 
  155.  155      * 
  156.  156      * NOTE: This is called automatically upon object instantiation; you should not need to call this directly.
  157.  157      */
  158.  158     function Connect() {
  159.  159         error_reporting(0);
  160.  160 
  161.  161         if ( $this->debugging ) echo "mpd->Connect() / host: ".$this->host.", port: ".$this->port."\n";
  162.  162         $this->mpd_sock = @fsockopen($this->host,$this->port,$errNo,$errStr,10);
  163.  163         if (!$this->mpd_sock) {
  164.  164             $this->errStr = "Socket Error: $errStr ($errNo)";
  165.  165             return NULL;
  166.  166         } else {
  167.  167             while(!feof($this->mpd_sock)) {
  168.  168                 $response =  fgets($this->mpd_sock,1024);
  169.  169                 if (strncmp(MPD_RESPONSE_OK,$response,strlen(MPD_RESPONSE_OK)) == 0) {
  170.  170                     $this->connected = TRUE;
  171.  171                     return $response;
  172.  172                     break;
  173.  173                 }
  174.  174                 if (strncmp(MPD_RESPONSE_ERR,$response,strlen(MPD_RESPONSE_ERR)) == 0) {
  175.  175                     $this->errStr = "Server responded with: $response";
  176.  176                     return NULL;
  177.  177                 }
  178.  178             }
  179.  179             // Generic response
  180.  180             $this->errStr = "Connection not available";
  181.  181             return NULL;
  182.  182         }
  183.  183     }

  185.  185     /* SendCommand()
  186.  186      * 
  187.  187      * Sends a generic command to the MPD server. Several command constants are pre-defined for 
  188.  188      * use (see MPD_CMD_* constant definitions above). 
  189.  189      */
  190.  190     function SendCommand($cmdStr,$arg1 = "",$arg2 = "") {
  191.  191         if ( $this->debugging ) echo "mpd->SendCommand() / cmd: ".$cmdStr.", args: ".$arg1." ".$arg2."\n";
  192.  192         if ( ! $this->connected ) {
  193.  193             echo "mpd->SendCommand() / Error: Not connected\n";
  194.  194         } else {
  195.  195             // Clear out the error String
  196.  196             $this->errStr = "";
  197.  197             $respStr = "";
  198.  198 
  199.  199             // Check the command compatibility:
  200.  200             if ( ! $this->_checkCompatibility($cmdStr) ) {
  201.  201                 return NULL;
  202.  202             }
  203.  203 
  204.  204             if (strlen($arg1) > 0) $cmdStr .= " \"$arg1\"";
  205.  205             if (strlen($arg2) > 0) $cmdStr .= " \"$arg2\"";
  206.  206             fputs($this->mpd_sock,"$cmdStr\n");
  207.  207             while(!feof($this->mpd_sock)) {
  208.  208                 $response = fgets($this->mpd_sock,1024);
  209.  209 
  210.  210                 // An OK signals the end of transmission -- we'll ignore it
  211.  211                 if (strncmp(MPD_RESPONSE_OK,$response,strlen(MPD_RESPONSE_OK)) == 0) {
  212.  212                     break;
  213.  213                 }
  214.  214 
  215.  215                 // An ERR signals the end of transmission with an error! Let's grab the single-line message.
  216.  216                 if (strncmp(MPD_RESPONSE_ERR,$response,strlen(MPD_RESPONSE_ERR)) == 0) {
  217.  217                     list ( $junk, $errTmp ) = split(MPD_RESPONSE_ERR . " ",$response );
  218.  218                     $this->errStr = strtok($errTmp,"\n");
  219.  219                 }
  220.  220 
  221.  221                 if ( strlen($this->errStr) > 0 ) {
  222.  222                     return NULL;
  223.  223                 }
  224.  224 
  225.  225                 // Build the response string
  226.  226                 $respStr .= $response;
  227.  227             }
  228.  228             if ( $this->debugging ) echo "mpd->SendCommand() / response: '".$respStr."'\n";
  229.  229         }
  230.  230         return $respStr;
  231.  231     }

  233.  233     /* QueueCommand() 
  234.  234      *
  235.  235      * Queues a generic command for later sending to the MPD server. The CommandQueue can hold 
  236.  236      * as many commands as needed, and are sent all at once, in the order they are queued, using 
  237.  237      * the SendCommandQueue() method. The syntax for queueing commands is identical to SendCommand(). 
  238.  238      */
  239.  239     function QueueCommand($cmdStr,$arg1 = "",$arg2 = "") {
  240.  240         if ( $this->debugging ) echo "mpd->QueueCommand() / cmd: ".$cmdStr.", args: ".$arg1." ".$arg2."\n";
  241.  241         if ( ! $this->connected ) {
  242.  242             echo "mpd->QueueCommand() / Error: Not connected\n";
  243.  243             return NULL;
  244.  244         } else {
  245.  245             if ( strlen($this->command_queue) == 0 ) {
  246.  246                 $this->command_queue = MPD_CMD_START_BULK . "\n";
  247.  247             }
  248.  248             if (strlen($arg1) > 0) $cmdStr .= " \"$arg1\"";
  249.  249             if (strlen($arg2) > 0) $cmdStr .= " \"$arg2\"";
  250.  250 
  251.  251             $this->command_queue .= $cmdStr ."\n";
  252.  252 
  253.  253             if ( $this->debugging ) echo "mpd->QueueCommand() / return\n";
  254.  254         }
  255.  255         return TRUE;
  256.  256     }

  259.  259      *

  261.  261      */
  262.  262     function SendCommandQueue() {
  263.  263         if ( $this->debugging ) echo "mpd->SendCommandQueue()\n";
  264.  264         if ( ! $this->connected ) {
  265.  265             echo "mpd->SendCommandQueue() / Error: Not connected\n";
  266.  266             return NULL;

  268.  268             $this->command_queue .= MPD_CMD_END_BULK . "\n";
  269.  269             if ( is_null($respStr = $this->SendCommand($this->command_queue)) ) {
  270.  270                 return NULL;
  271.  271             } else {
  272.  272                 $this->command_queue = NULL;
  273.  273                 if ( $this->debugging ) echo "mpd->SendCommandQueue() / response: '".$respStr."'\n";
  274.  274             }
  275.  275         }
  276.  276         return $respStr;
  277.  277     }

  279.  279     /* AdjustVolume() 
  280.  280      *
  281.  281      * Adjusts the mixer volume on the MPD by <modifier>, which can be a positive (volume increase),
  282.  282      * or negative (volume decrease) value. 
  283.  283      */
  284.  284     function AdjustVolume($modifier) {
  285.  285         if ( $this->debugging ) echo "mpd->AdjustVolume()\n";
  286.  286         if ( ! is_numeric($modifier) ) {
  287.  287             $this->errStr = "AdjustVolume() : argument 1 must be a numeric value";

  293.  293         $ret = $this->SetVolume($newVol);
  294.  294

  297.  297     }
  298.  298 
  299.  299     /* SetVolume() 
  300.  300      *
  301.  301      * Sets the mixer volume to <newVol>, which should be between 1 - 100.
  302.  302      */
  303.  303     function SetVolume($newVol) {
  304.  304         if ( $this->debugging ) echo "mpd->SetVolume()\n";
  305.  305         if ( ! is_numeric($newVol) ) {
  306.  306             $this->errStr = "SetVolume() : argument 1 must be a numeric value";
  307.  307             return NULL;
  308.  308         }
  309.  309 
  310.  310         // Forcibly prevent out of range errors
  311.  311         if ( $newVol < 0 )   $newVol = 0;
  312.  312         if ( $newVol > 100 ) $newVol = 100;
  313.  313 
  314.  314         // If we're not compatible with SETVOL, we'll try adjusting using VOLUME
  315.  315         if ( $this->_checkCompatibility(MPD_CMD_SETVOL) ) {
  316.  316             if ( ! is_null($ret = $this->SendCommand(MPD_CMD_SETVOL,$newVol))) $this->volume = $newVol;
  317.  317         } else {
  318.  318             $this->RefreshInfo();     // Get the latest volume
  319.  319             if ( is_null($this->volume) ) {
  320.  320                 return NULL;
  321.  321             } else {
  322.  322                 $modifier = ( $newVol - $this->volume );
  323.  323                 if ( ! is_null($ret = $this->SendCommand(MPD_CMD_VOLUME,$modifier))) $this->volume = $newVol;
  324.  324             }
  325.  325         }
  326.  326 
  327.  327         if ( $this->debugging ) echo "mpd->SetVolume() / return\n";
  328.  328         return $ret;
  329.  329     }

  348.  348      * of tracks to add, to the end of the playlist. This is used to add many, many tracks to

  356.  356         }

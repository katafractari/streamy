  329. 329  while(radioForm.tagName!="FORM"){
  330. 330  radioForm = radioForm.parentNode;
  331. 331  tmpCnt++;
  332. 332  if(tmpCnt>50){
  333. 333  window.alert("encountered javascript error\n[parentNode]")
  334. 334  break;
  335. 335  }
  336. 336  }
  337. 337  inputs=radioForm.getElementsByTagName("input");
  338. 338  for(i=0;i<inputs.length;i++){
  339. 339  if(inputs[i].type=="radio"){
  340. 340  //IDs look like this:  realId: blah    fakeId: fakeblah
  341. 341  if(inputs[i].name==realRadio.name){ //is part of the same radio group, uncheck it.
  342. 342  inputs[i].checked=false; //uncheck the actual button
  343. 343  document.getElementById("fake"+inputs[i].id).src=imageRadioUnchecked;
  344. 344  if(inputs[i].id==realRadioId){
  345. 345  inputs[i].checked=true; //check the actual button
  346. 346  document.getElementById("fake"+inputs[i].id).src=imageRadioChecked;
  347. 347  }
  348. 348  }
  349. 349  }
  350. 350  }

  442. 442  //function to trigger events that are built into the form elements that have been hidden
  443. 443  function triggerEvent(obj, evt){
  444. 444  if(document.all){
  445. 445  if(evt=="click"){
  446. 446  res = obj.fireEvent("onclick");
  447. 447  }else if(evt=="change"){
  448. 448  res = obj.fireEvent("onchange");
  449. 449  }
  450. 450  }else{
  451. 451  //NOTE - in the mozilla event model, I am cancelling the bubbleUp!  (1st false)
  452. 452  // this is needed to prevent odd interaction, but could cause other issues!
  453. 453  if(evt=="click"){
  454. 454  mouseEvent = document.createEvent('MouseEvents');
  455. 455  mouseEvent.initMouseEvent('click',true,true,window,1,0,0,0,0,false,false,false,false,0,null);
  456. 456  obj.dispatchEvent(mouseEvent); 
  457. 457  }else if(evt=="change"){
  458. 458  mouseEvent = document.createEvent('HTMLEvents');
  459. 459  mouseEvent.initEvent('change',true,true,window,1,0,0,0,0,false,false,false,false,0,null);
